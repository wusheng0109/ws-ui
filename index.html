# 快速开始

# 安装组件库

npm i ws0909-ui

# 引用组件库

在main.js中引用组件库

//全部引入
import 'ws0909-ui/dist/css/index.css'
import WUI from 'ws0909-ui'
Vue.use(WUI)

//按需引入
import  'ws0909-ui/dist/css/demo.scss'
import { Demo } from 'ws0909-ui'

Vue.use(Demo)
